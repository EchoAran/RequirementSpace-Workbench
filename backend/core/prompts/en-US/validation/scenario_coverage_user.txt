Return this JSON shape:
{{

    "repair_type": "reassign_scenario_feature",
    "title": "Reassign {scenario_name} to target feature",
    "rationale": "Why the content matches",
    "confidence": 0.85,
    "patch": {{"updateNodes": [{{"kind": "scenario", "id": {scenario_id}, "feature_id": {feature_id}}]}},
    "requires_user_decision": false
  }}],
  "fallback": {{"kind": "generation_draft", "reason": "No existing scenario can be reused"}}
}}
